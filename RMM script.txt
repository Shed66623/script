local character, hrp

local function attachCharacter(char)
    character = char
    local humanoid = char:WaitForChild("Humanoid")
    hrp = char:WaitForChild("HumanoidRootPart")

    pcall(function()
        local temp = char:FindFirstChild("Temperature")
        if temp then temp:Destroy() end
    end)

    humanoid.Died:Connect(function()
        hrp = nil
    end)
end

if player.Character then
    attachCharacter(player.Character)
end
player.CharacterAdded:Connect(attachCharacter)

local function tpFreeze(cf, duration, cd)
    local conn = RunService.Heartbeat:Connect(function()
        if hrp and hrp.Parent then
            hrp.CFrame = cf
        end
    end)

    local clicking = true
    task.spawn(function()
        while clicking do
            if cd then
                pcall(function() fireclickdetector(cd) end)
            end
            task.wait(0.01)
        end
    end)

    task.wait(duration)
    clicking = false
    if conn then conn:Disconnect() end
end

task.spawn(function()
    task.wait(11)
    local vr = workspace:FindFirstChild("VoiceRecorder")
    local vrClick = vr and vr:FindFirstChild("ClickDetector")
    if vrClick then
        tpFreeze(CFrame.new(-240, 82, 56), 7, vrClick)
    end

    local pi = workspace:FindFirstChild("PunchIn")
    local piClick = pi and pi:FindFirstChild("ClickDetector")
    if piClick then
        tpFreeze(CFrame.new(-296, 82, 16), 80, piClick)
    end

    getgenv().TP = true
    getgenv().Noclip = true
end)
